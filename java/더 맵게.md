풀이 [1차 오답]

import java.util.PriorityQueue;

class Solution {
    public int solution(int[] scoville, int K) {
        int answer = 0;
        PriorityQueue<Integer> heap = new PriorityQueue<Integer>();
        
        for(int i =0; i<scoville.length ; i++){
            heap.add(scoville[i]);
        }
        while(heap.peek() < K){
            int min1 = heap.poll();
            int min2 = heap.poll();
            int a = min1 + (min2*2);
            answer++; 
            if(heap.peek() < K ){
                heap.add(a);
            }
            else {
                break;
            }
        }
        return answer;
    }
}
